class Result
{
    public static int getTotalX(List<int> a, List<int> b)
    {
        List<int> nums = [];
        int count = 0;
        
        for (int i = a.Max(); i <= b.Min(); i++) {
            bool isValid = true;
            
            foreach (var item in a)
                if (i % item != 0) 
                    isValid = false;
                    
            if (isValid)
                nums.Add(i);
        }
        
        for (int i = 0; i < nums.Count; i++) {
            bool isValid = true;
            
            foreach (var item in b)
                if (item % nums[i] != 0)
                    isValid = false;
            
            if (isValid)
                count++;
        }
        
        return count;
    }
}
